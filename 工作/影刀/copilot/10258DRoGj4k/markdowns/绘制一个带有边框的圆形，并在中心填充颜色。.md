```python
# 打开绘图软件
process_info=xbot_visual.desktop.launch(process_name="绘图软件")
# 点击圆形工具
xbot_visual.desktop.click(window="0", element="圆形工具")
# 获取画布边框
bounding = xbot_visual.desktop.get_attribute(window="0", element="画布", attribute_name="bound")
# 画布中心点横纵坐标
center_x,center_y = bounding.center_x,bounding.center_y
# 将鼠标移动中心点
xbot_visual.mouse.move(x=center_x, y=center_y)
# 按下鼠标
xbot_visual.mouse.click(click_type="down")
# 移动鼠标到center_x+100，center_y+100
xbot_visual.mouse.move(x=center_x+100, y=center_y+100)
# 松开鼠标,完成绘制
xbot_visual.mouse.click(click_type="up")
# 点击填充工具
xbot_visual.desktop.click(window="0", element="填充工具")
# 点击图标中心，完成填充
xbot_visual.desktop.click(window="0", element="图标中心")
```
