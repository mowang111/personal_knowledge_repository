import time

xbot_visual.desktop.input(window="0", element="文件路径输入框", text="C:/Videos/movie.mp4")
# 点击确认打开
xbot_visual.desktop.click(window="0", element="打开")
# 点击播放按钮
xbot_visual.desktop.click(window="0", element="播放")
# 休眠一段时间，让视频播到一半
time.sleep(10)
# 点击暂停按钮
xbot_visual.desktop.click(window="0", element="暂停")
# 调整播放速度为2倍速
xbot_visual.desktop.select_by_index(window="0", element="播放速度", index="2")

xbot_visual.desktop.click(window="0", element="播放")
