```python
# 打开订单处理软件
process_info=xbot_visual.desktop.launch(process_name="订单处理软件")
# 点击进入当天订单页面
xbot_visual.desktop.click(window="0", element="当天订单")
# 遍历订单对象列表
for order in xbot_visual.desktop.iter_all_elements(window="0", element="订单对象"):
    # 获取订单对象的属性，如订单日期
    order_date = xbot_visual.desktop.get_attribute(window="0", element=order, attribute_name="text")
    # 判断订单日期是否为当天
    if order_date == "当天":
        # 点击进入订单详情
        xbot_visual.desktop.click(window="0", element=order)
        # 获取订单详情中的退货请求对象列表
        return_request_list = xbot_visual.desktop.find_elements(window="0", element="退货请求")
        # 判断是否有退货请求
        if return_request_list.length > 0:
            # 标记订单，如勾选订单
            xbot_visual.desktop.click(window="0", element="订单勾选框")
            # 根据订单处理软件窗口名称获取订单处理软件窗口对象
            order_software_window = xbot_visual.desktop.get(title="订单处理软件")
            # 移动窗口到新的位置
            xbot_visual.desktop.move(window=order_software_window, x=0, y=0)
            # 结束遍历
            break
        # 没有退货请求
        else:
            # 返回订单列表
            xbot_visual.desktop.click(window="0", element="返回订单列表")
    # 没有当天有退货请求的订单
    else:
        # 输出提示信息
        print("没有当天有退货请求的订单")
```
