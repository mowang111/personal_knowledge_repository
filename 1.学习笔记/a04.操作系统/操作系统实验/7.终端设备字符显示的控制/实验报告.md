keyboard.S中，添加一个`f12state`byte来存储f12的状态
```c
.globl f12state
f12state: .byte 0

func:
	...
	...
	cmpb $11, %al
	ja end_func
	notb f12state // 这一句是添加的
	...
```

tty_io中在`tty_write`中添加改变字母为`*`的代码
```c
while (nr>0) {
		sleep_if_full(&tty->write_q);
		if (current->signal)
			break;
		while (nr>0 && !FULL(tty->write_q)) {
			c=get_fs_byte(b);
			if (O_POST(tty)) {
				if (c=='\r' && O_CRNL(tty))
					c='\n';
				else if (c=='\n' && O_NLRET(tty))
					c='\r';
				if (c=='\n' && !cr_flag && O_NLCR(tty)) {
					cr_flag = 1;
					PUTCH(13,tty->write_q);
					continue;
				}
				if (O_LCUC(tty))
					c=toupper(c);
			}
			b++; nr--;
			cr_flag = 0;
			if(f12state){
				if(c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z'){
					c = '*';
				}
			}
			PUTCH(c,tty->write_q);
		}
		tty->write(tty);
		if (nr>0)
			schedule();
	}
	return (b-buf);
}
```
